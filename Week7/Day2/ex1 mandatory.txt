CREATE DATABASE public;
CREATE TABLE items (
    id INT PRIMARY KEY,
    name VARCHAR(100),
    price INT
);
CREATE TABLE customers (
    id INT PRIMARY KEY,
    first_name VARCHAR(100),
    last_name VARCHAR(100)
);
